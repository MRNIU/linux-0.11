7 /*
8  * 'buffer.c' implements the buffer-cache functions. Race-conditions have
9  * been avoided by NEVER letting a interrupt change a buffer (except for the
10  * data, of course), but instead letting the caller do it. NOTE! As interrupts
11  * can wake up a caller, some cli-sti sequences are needed to check for
12  * sleep-on-calls. These should be extremely quick, though (I hope).

102 /*
103  * This routine checks whether a floppy has been changed, and
104  * invalidates all buffer-cache-entries in that case. This
105  * is a relatively slow routine, so we have to try to minimize using
106  * it. Thus it is called only upon a 'mount' or 'open'. This
107  * is the best way of combining speed and utility, I think.
108  * People changing diskettes in the middle of an operation deserve
109  * to loose :-)
110  *
111  * NOTE! Although currently this is only for floppies, the idea is
112  * that any additional removable block-device will use this routine,
113  * and that mount/open needn't know that floppies/whatever are
114  * special.
115  */
116 void check_disk_change(int dev) {
117  int i;
118 
119  if (MAJOR(dev) != 2)
120  return;

124  if (super_block[i].s_dev == dev)

133 static inline void remove_from_queues(struct buffer_head *bh) {
134 /* remove from hash-queue */
135  if (bh->b_next)
136  bh->b_next->b_prev = bh->b_prev;
137  if (bh->b_prev)
138  bh->b_prev->b_next = bh->b_next;
139  if (hash(bh->b_dev, bh->b_blocknr) == bh)
140  hash(bh->b_dev, bh->b_blocknr) = bh->b_next;
141 /* remove from free list */
142  if (!(bh->b_prev_free) || !(bh->b_next_free))
143  panic("Free block list corrupted");
144  bh->b_prev_free->b_next_free = bh->b_next_free;
145  bh->b_next_free->b_prev_free = bh->b_prev_free;
146  if (free_list == bh)
147  free_list = bh->b_next_free;
148 }

150 static inline void insert_into_queues(struct buffer_head *bh) {
151 /* put at end of free list */
152  bh->b_next_free = free_list;
153  bh->b_prev_free = free_list->b_prev_free;
154  free_list->b_prev_free->b_next_free = bh;
155  free_list->b_prev_free = bh;
156 /* put the buffer in new hash-queue if it has a device */
157  bh->b_prev = NULL;
158  bh->b_next = NULL;
159  if (!bh->b_dev)
160  return;
161  bh->b_next = hash(bh->b_dev, bh->b_blocknr);
162  hash(bh->b_dev, bh->b_blocknr) = bh;
163  bh->b_next->b_prev = bh;
164 }

175 /*
176  * Why like this, I hear you say... The reason is race-conditions.
177  * As we don't lock buffers (unless we are readint them, that is),
178  * something might happen to it while we sleep (ie a read-error
179  * will force it bad). This shouldn't really happen currently, but
180  * the code is ready.
181  */
182 struct buffer_head *get_hash_table(int dev, int block) {
183  struct buffer_head *bh;
184 
185  for (;;) {
186  if (!(bh = find_buffer(dev, block)))
187  return NULL;
188  bh->b_count++;
189  wait_on_buffer(bh);
190  if (bh->b_dev == dev && bh->b_blocknr == block)
191  return bh;
192  bh->b_count--;
193  }
194 }

197  * Ok, this is getblk, and it isn't very clear, again to hinder
198  * race-conditions. Most of the code is seldom used, (ie repeating),
199  * so it should be much more efficient than it looks.
200  *
201  * The algoritm is changed: hopefully better, and an elusive bug removed.

203 #define BADNESS(bh) (((bh)->b_dirt<<1)+(bh)->b_lock)

205 struct buffer_head *getblk(int dev, int block) {
206  struct buffer_head *tmp, *bh;
207 
208  repeat:
209  if ((bh = get_hash_table(dev, block)))
210  return bh;
211  tmp = free_list;
212  do {
213  if (tmp->b_count)
214  continue;
215  if (!bh || BADNESS(tmp) < BADNESS(bh)) {
216  bh = tmp;
217  if (!BADNESS(tmp))
218  break;
219  }
220 /* and repeat until we find something good */
221  } while ((tmp = tmp->b_next_free) != free_list);
222  if (!bh) {
223  sleep_on(&buffer_wait);
224  goto repeat;
225  }
226  wait_on_buffer(bh);
227  if (bh->b_count)
228  goto repeat;
229  while (bh->b_dirt) {
230  sync_dev(bh->b_dev);
231  wait_on_buffer(bh);
232  if (bh->b_count)
233  goto repeat;
234  }
235 /* NOTE!! While we slept waiting for this block, somebody else might */
236 /* already have added "this" block to the cache. check it */
237  if (find_buffer(dev, block))
238  goto repeat;
239 /* OK, FINALLY we know that this buffer is the only one of it's kind, */
240 /* and that it's unused (b_count=0), unlocked (b_lock=0), and clean */
241  bh->b_count = 1;
242  bh->b_dirt = 0;
243  bh->b_uptodate = 0;
244  remove_from_queues(bh);
245  bh->b_dev = dev;
246  bh->b_blocknr = block;
247  insert_into_queues(bh);
248  return bh;
249 }

286 /*
287  * bread_page reads four buffers into memory at the desired address. It's
288  * a function of its own, as there is some speed to be got by reading them
289  * all at the same time, not waiting for one to be read, and then another
290  * etc.
291  */
292 void bread_page(unsigned long address, int dev, int b[4]) {
293  struct buffer_head *bh[4];
294  int i;
295 
296  for (i = 0; i < 4; i++)
297  if (b[i]) {
298  if ((bh[i] = getblk(dev, b[i])))
299  if (!bh[i]->b_uptodate)
300  ll_rw_block(READ, bh[i]);
301  } else
302  bh[i] = NULL;
303  for (i = 0; i < 4; i++, address += BLOCK_SIZE)
304  if (bh[i]) {
305  wait_on_buffer(bh[i]);
306  if (bh[i]->b_uptodate)
307  COPYBLK((unsigned long) bh[i]->b_data, address);
308  brelse(bh[i]);
309  }
310 }

362  h->b_next_free = h + 1;
363  h++;
