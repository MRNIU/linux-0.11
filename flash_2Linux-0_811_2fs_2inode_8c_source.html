151 {
152  if (!inode)
153  return;

155  if (!inode->i_count)
156  panic("iput: trying to free free inode");

172  sync_dev(inode->i_zone[0]);
173  wait_on_inode(inode);
174  }
175 repeat:
176  if (inode->i_count>1) {
177  inode->i_count--;

179  }

187  wait_on_inode(inode);
