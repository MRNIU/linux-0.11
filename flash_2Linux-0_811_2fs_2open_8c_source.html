48 {
49  struct m_inode * inode;
50  int res, i_mode;
51 
52  mode &= 0007;
53  if (!(inode=namei(filename)))
54  return -EACCES;
55  i_mode = res = inode->i_mode & 0777;
56  iput(inode);
57  if (current->uid == inode->i_uid)
58  res >>= 6;
59  else if (current->gid == inode->i_gid)
60  res >>= 6;
61  if ((res & 0007 & mode) == mode)
62  return 0;
63  /*
64  * XXX we are doing this test last because we really should be
65  * swapping the effective with the real user id (temporarily),
66  * and then calling suser() routine. If we do call the
67  * suser() routine, it needs to be called last.
68  */
69  if ((!current->uid) &&
70  (!(mode & 1) || (i_mode & 0111)))
71  return 0;
72  return -EACCES;
73 }

139 {

143

151  f=0+file_table;
152  for (i=0 ; i<NR_FILE ; i++,f++)
153  if (!f->f_count) break;

155  return -EINVAL;

158  current->filp[fd]=NULL;

160  return i;

162  //printf("inode_dev=%u", inode->i_dev);
163 /* ttys are somewhat special (ttyxx major==4, tty major==5) */
164  if (S_ISCHR(inode->i_mode)) {
165  if (MAJOR(inode->i_zone[0])==4) {
166  if (current->leader && current->tty<0) {
167  current->tty = MINOR(inode->i_zone[0]);
168  tty_table[current->tty].pgrp = current->pgrp;
169  }

197

199  return -EINVAL;
200  current->close_on_exec &= ~(1<<fd);
201  if (!(filp = current->filp[fd]))
202  return -EINVAL;
203  current->filp[fd] = NULL;
204  if (filp->f_count == 0)
205  panic("Close: file count is 0");
206  if (--filp->f_count)
