476 // 软盘读写请求项处理函数
477 void do_fd_request(void){
478  unsigned int block;
479 
480  seek = 0;
481  if(reset){ // 如果复位标志已置位，则执行软盘复位操作，并返回
482  reset_floppy();
483  return;
484  }
485  if(recalibrate){ // 如果重新校正标志已置位，则执行软盘重新校正操作，并返回
486  recalibrate_floppy();
487  return;
488  }
489  INIT_REQUEST; // 检测请求项的合法性(参见 kernel/blk_drv/blk.h 123)
490 // 将请求项结构中软盘设备号中的软盘类型(MINOR(CURRENT->dev)>>2)作为索引取得软盘参数块
491  floppy = (MINOR(CURRENT->dev)>>2) + floppy_type;
492 // 如果当前驱动器不是请求项中指定的驱动器，则置标志 seek，标志需要进行寻道操作。然后置请求项设备
493 // 为当前驱动器
494  if(current_drive != CURRENT_DEV) seek = 1;
495  current_drive = CURRENT_DEV;
496 // 设置读写起始扇区。因为每次读写是以块为单位(1 块 2 个扇区)，所以起始扇区需要起码比磁盘总扇区数
497 // 小 2 个扇区。否则结束该次软盘请求项，执行下一个请求项
498  block = CURRENT->sector; // 去当前软盘请求项中起始扇区号-->block
499  if(block+2 > floppy->size){ // 如果 block+2 大于磁盘扇区总数，则
500  end_request(0); // 结束本次软盘请求项
501  goto repeat;
502  }
503 // 球队赢才磁道上的扇区号，磁头号，磁道号，搜寻磁道号(对于软驱读不同格式的盘)
504  sector = block % floppy->sect; // 起始扇区对每磁道扇区数取模，得磁道上扇区号
505  block /= floppy->sect; // 起始扇区对每磁道扇区数取整，得起始磁道数
506  head = block % floppy->head; // 起始磁道数对磁头数取模，得操作的磁头号
507  track = block / floppy->head; // 起始磁道数对磁头数取整，得操作的磁道号
508  seek_track = track << floppy->stretch; // 相应于驱动器中盘类型进行调整，的寻道号。
509 // 如果寻道号与当前磁头所在磁道不同，则置需要寻道标志 seek
510  if(seek_track != current_track) seek = 1;
511  sector++; // 磁盘上世纪扇区计数是从 1 算起
512  if(CURRENT->cmd == READ) // 如果请求项中是读操作，则置软盘读命令码
513  command = FD_READ;
514  else if(CURRENT->cmd == WRITE) // 如果请求项中是写操作，则置软盘写命令码
515  command = FD_WRITE;
516  else
517  panic("do_fd_request: unknown command");
518 // 添加定时器，用于指定驱动器到能正常运行所诉掩饰的时间(滴答数)，当定时时间到就调用函数 floppy_on_interrupt().
519  add_timer(ticks_to_floppy_on(current_drive),&floppy_on_interrupt);
520 }
521 // 软盘系统初始化。
522 // 设置软盘块设备的请求处理函数(do_fd_request()),并设置软盘中断门(int 0x26,对应硬件中断请
523 // 求信号 IRQ6)，然后取消对该中断信号的屏蔽，允许软盘控制器 FDC 发送中断请求信号

528 }
