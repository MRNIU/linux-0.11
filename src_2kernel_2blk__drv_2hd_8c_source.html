106  /*
107  We querry CMOS about hard disks : it could be that we have a SCSI/ESDI/etc
108  controller that is BIOS compatable with ST-506,and thus showing up in our BIOS
109  table,but not register compatable,and therefore not present in CMOS.
110 
111  Furthurmore,we will assume that our ST-506 drives <if any> are the primary
112  drives in the system,and the ones reflected as drive 1 or 2.
113 
114  The first drive is stored in the high nibble of CMOS byte 0x12,the second
115  in the low nibble. This will be either a 4 bit drive type or 0xf indicating
116  use byte 0x19 for an 8 bit type,drive 1 ,0x1a for drive 2 in CMOS.
117 
118  Needless to say,a non-zero value means we have an AT controller hard disk
119  for that drive.
120  */
121  /*
122  我们对 CMOS 有关硬盘的信息有些怀疑：可能会出现这样的情况，我们有一块 SCSI/ESDI 等的控制器，
123  它是以 ST-506 方式与 BIOS 兼容的，因而会出现在我们的 BIOS 参数表中，但又不是寄存器兼容的，
124  因此这些参数在 CMOS 中不存在。
125  另外，我们假设 ST-506 驱动器(如果有的话)是系统中的基本驱动器，即以驱动器 1 或 2 出现的驱动器。
126  第 1 个驱动器参数存放在 CMOS 字节 0x12 的高半字节中，第 2 个存放在低半字节中。该 4 位字节信息
127  可以是驱动器类型，也可能仅是 0xf.0xf 表示使用 CMOS 中 0x19 字节作为驱动器 1 的 8 位类型字节，
128  使用 CMOS 中 0x1A 字节作为驱动器 2 的类型字节。
129  总之，一个非零值意味着我们有一个 AT 控制器硬盘兼容的驱动器。

197

260 // 读操作中断调用函数。将在执行硬盘中断处理程序中被调用。
261 static void read_intr(void){
262  if(win_result()){ // 若控制器忙、读写错或命令执行错，则进行读写硬盘失败处理，然后再次请求
263  bad_rw_intr(); // 硬盘作相应(复位)处理。
264  do_hd_request();
265  return;
266  }
267  port_read(HD_DATA,CURRENT->buffer,256); // 将数据从数据寄存器口读到请求结构缓冲。
268  CURRENT->errors=0; // 清出错次数。
269  CURRENT->buffer+=512; // 调整缓冲区指针，指向新的空区
270  CURRENT->sector++; // 起始扇区号加 1
271  if(--CURRENT->nr_sectors){ // 如果所需读出的扇区数还没有读完，则再次置硬盘调用 C 函数指针
272  do_hd=&read_intr; // 为 read_intr().因为硬盘中断处理程序每次调用 do_hd 时都会将该函数
273  return; // 指针置空。参见 system_call.s
274  }
275  end_request(1); // 若全部扇区数据已经读完，则处理请求结束事宜
276  do_hd_request(); // 执行其它硬盘请求操作
277 }

296 }

313  dev=MINOR(CURRENT->dev); // CURRENT 定义为(blk_dev[MAJOR_NR].current_request)

315 // 如果子设备号不存在或者起始扇区大于该分区扇区数 -2，则结束该请求，并跳转到标号 repeat 处
316 // (定义在 INIT_REQUEST 开始处)。因为一次要求读写 2 个扇区(512*2 B),所以请求的扇区号不能
317 // 大于分区中倒数第二个扇区号。
318  if(dev>=5*NR_HD||block+2>hd[dev].nr_sects){

320  goto repeat; // 该标号在 blk.h 最后面
321  }
322  block += hd[dev].start_sect; // 将所需读的块对应到整个硬盘上的绝对扇区号

324 // 下面嵌入汇编代码用来从硬盘信息结构中根据起始扇区号和每磁道扇区数计算在磁道中的扇区号(sec)、
325 // 所在柱面号(cyl)、和磁头号(head)
326  __asm__("divl %4":"=a"(block),"=d"(sec):"0"(block),"1"(0),"r"(hd_info[dev].sect));
327  __asm__("divl %4":"=a"(cyl),"=d"(head):"0"(block),"1"(0),"r"(hd_info[dev].head));
328  sec++;
329  nsect=CURRENT->nr_sectors; // 欲读、写的扇区数
330  if(reset){ // 如果 reset 置 1，则执行复位操作。
331  reset=0; // 复位硬盘和控制器，并置需要重新校正，返回
332  recalibrate=1;
333  reset_hd(CURRENT_DEV);
334  return;
335  }
336  if(recalibrate){ // 如果重新校正标志(recalibrate)置位，则首先复位该标志，然后向硬盘控制器
337  recalibrate=0; // 发送重新校正命令。
338  hd_out(dev,hd_info[CURRENT_DEV].sect,0,0,0,WIN_RESTORE,&recal_intr);
339  return;
340  }
341 // 如果当前请求是写扇区操作，则发送写命令，循环读取状态寄存器信息并判断请求服务标志 DRQ_STAT 是
342 // 否置位。DRQ_STAT 是硬盘状态寄存器的请求服务位(include/linux/hdreg.h 222)
343  if(CURRENT->cmd==WRITE){
344  hd_out(dev,nsect,sec,head,cyl,WIN_WRITE,&write_intr);
345  for(i=0;i<3000&&!(r=inb_p(HD_STATUS)&DRQ_STAT);i++)
346  // nothing
347 // 如果请求服务置位则退出循环。若等到循环结束也没有置位，则此次写硬盘操作失败，去处理下一个硬盘
348 // 请求。否则向硬盘控制器数据寄存器端口 HD_DATA 写入 1 个扇区的数据。
349  if(!r){
350  bad_rw_intr();
351  goto repeat; // 该标号在 blk.h 最后面，即跳到 311 行。
352  }
353  port_write(HD_DATA,CURRENT->buffer,256);
354 // 如果当前请求是读硬盘扇区，则向硬盘控制器发送读扇区命令
355  } else if(CURRENT->cmd==READ){
356  hd_out(dev,nsect,sec,head,cyl,WIN_READ,&read_intr);
357  } else
358  panic("unknown hd-command");
359 }

366  // 中断请求信号。
367 }
