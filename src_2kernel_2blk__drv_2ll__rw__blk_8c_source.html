58 // it can muck with the request-lists in peace.
59 // add-request() 向链表中加入一项请求。它关闭中断，这样就能安全地处理请求链表了。
60 // 向链表中加入请求项。参数 dev 指定块设备，req 是请求的结构信息。
61 static void add_request(struct blk_dev_struct * dev,struct request * req){
62  struct request * tmp;
63  req->next=NULL;
64  cli(); // 关中断
65  if(req->bh)
66  req->bh->b_dirt=0;
67 // 如果 dev 的当前请求(current_request)子段为空，则表示目前该设备没有请求项，本次是第 1 个
68 // 请求项，因此可将块设备当前请求指针直接指向请求项，并立刻执行相应设备的请求函数。
69  if(!(tmp=dev->current_request)){
70  dev->current_request=req;
71  sti(); // 开中断
72  (dev->request_fn)(); // 执行设备请求函数，对于硬盘(3)是 do_hd_request()
73  return;
74  }
75 // 若该设备已有请求项在等待，则首先利用电梯算法搜索最佳位置，然后将当前请求插入请求链表中。
76  for(;tmp->next;tmp=tmp->next)
77  if((IN_ORDER(tmp,req)||!IN_ORDER(tmp,tmp->next))&&IN_ORDER(req,tmp->next))
78  break;
79  req->next=tmp->next;
80  tmp->next=req;
81  sti();
82 }

84 static void make_request(int major,int rw,struct buffer_head * bh){
85  struct request * req;
86  int rw_ahead;
87 
88 // WRITEA/READA is special case - it is not really needed,so if the buffer is
89 // locked,we just forget about it,else it's a normal read
90 // WRITEA/READA 是特殊的情况 - 它们并不是必要的，所以如果缓冲区已经上锁，我们就不再管它而退出，
91 // 否则的话就执行一般的读/写操作
92 // 这里 "WRITEA" 和 "READA" 后面的 "A" 代表英文单词 Ahead， 表示提前预读/写数据块的意思。
93 // 当指定的缓冲区正在使用，已被上锁时，就放弃预读/写请求
94  if(rw_ahead=(rw==READA||rw==WRITEA)){
95  if(bh->b_lock) return;
96  if(rw==READA) rw=READ;
97  else rw=WRITE;
98  }
99 // 如果命令不是 READ 或 WRITE 则表示内核程序有错，显示出错信息并死机
100  if(rw!=READ&&rw!=WRITE)
101  panic("Bad block dev command,must be R/W/RA/WA");
102 // 锁定缓冲区，如果缓冲区已经上锁，则当前任务(进程)就会睡眠，直到被明确地唤醒
103  lock_buffer(bh);
104 // 如果命令是写并且缓冲区数据不脏，或者命令是读并且缓冲区数据是更新过时，则不用添加这个请求。
105 // 将缓冲区解锁并退出。
106  if((rw==WRITE&&!bh->b_dirt)||(rw==READ&&bh->b_uptodate)){
107  unlock_buffer(bh);
108  return;
109  }
110 repeat:
111 // we don't allow the write-requests to fill up the queue completely:
112 // we want some room for reads: they take precedence.The last third of the requests
113 // are only for reads.
114 // 我们不能让队列中全部都是写请求项：我们需要为读请求保留一些空间：读操作是优先的。请求队列的后
115 // 三分之一空间是为读准备的。
116 // 请求项是从请求数组末尾开始搜索空项填入的。根据上述要求，对于读命令请求，可以直接从队列末尾开始
117 // 操作，而写请求则只能从队列的 2/3 处向头上搜索空项填入
118  if(rw==READ)
119  req=request+NR_REQUEST; // 对于读请求，将队列指针指向队列尾部。
120  else
121  req=request+((NR_REQUEST*2)/3); // 对于写请求，队列指针指向队列 2/3 处
122 // find an empty request 搜索一个空请求项
123 // 从后向前搜索，当请求结构 request 的 dev 字段值 =-1 时，表示该项未被占用
124  while(--req>=request)
125  if(req->dev>0)
126  break;
127 // if none found,sleep on new requests:check for rw_ahead
128 // 如果没有找到空闲项，则让该次新请求睡眠：需检查是否提前读/写
129 // 如果没有一项是空闲的(此时 request 数组指针已经搜索越过头部)，则查看此次请求是不是提前读/写
130 // (READA 或 WRITEA)，如果是则放弃此次请求。否则让本次请求睡眠(等待请求队列腾出空项)，过一会
131 // 再来搜索请求队列
132  if(req<request){ // 如果请求队列中没有空项，则
133  if(rw_ahead){ // 如果是提前读/写请求，则解锁缓冲区，退出
134  unlock_buffer(bh);
135  return;
136  }
137  sleep_on(&wait_for_request); // 否则让本次请求睡眠，过会再查看请求队列
138  goto repeat;
139  }
140 // fill up the request-info,and add it to the queue
141 // 向空闲请求项中天蝎请求信息，并将其加入队列中
142 // 请求结构参见(kernel/blk_drv/blk.h 42)
143  req->dev=bh->b_dev; // 设备号
144  req->cmd=rw; // 命令(READ/WRITE)
145  req->errors=0; // 操作时产生的错误次数
146  req->sector=bh->b_blocknr<<1; // 起始扇区(1 块 = 2 扇区)
147  req->nr_sectors=2; // 读写扇区数
148  req->buffer=bh->b_data; // 数据缓冲区
149  req->waiting=NULL; // 任务等待操作执行完成的地方
150  req->bh=bh; // 缓冲区头指针
151  req->next=NULL; // 指向下一请求项
152  add_request(major+blk_dev,req); // 将请求项加入队列中(blk_dev[major],req)
153 }
