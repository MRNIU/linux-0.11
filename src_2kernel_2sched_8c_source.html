114  for(p=&LAST_TASK;p>&FIRST_TASK;--p)
115  if(*p){

117 // 置 SIGALRM 信号，然后清 alarm。jiffies 是系统从开机算起的滴答数(10ms/滴答)。定义在 sched.h
118 // 第 22222 行。
119  if((*p)->alarm && (*p)->alarm<jiffies){

122  }
123 // 如果信号位图中除被阻塞的信号外还有其他信号，并且任务处于可中断状态，则置任务为就绪状态。
