37 static char * number(char * str,int num,int base,int size,int precision,int type){
38  char c,sign,tmp[36];
39  const char * digits="0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ";
40  int i;
41 // 若类型 type 指出用小写字母，则定义小写字母集。若类型指出要左调整(靠左边界)，则屏蔽填零标志。
42 // 若进制基数小于 2 或大于 36，则退出处理，也即本程序只能处理基数在 2-32 之间的数
43  if(type & SMALL)digits="0123456789abcdefghijklmnopqrstuvwxyz";
44  if(type & LEFT) type &=~ZEROPAD;
45  if(base<2||base>36) return 0;
46 // 若类型指出要填零，则置字符变量 c='0' (即 ''),否则 c 等于空格字符。若类型指出是带符号数并且
47 // 数值 num 小于 0，则置符号变量 sign=负号，并使 num 取绝对值。否则如果类型指出是加号，
48 //则置 sign=加号，否则若类型带空格标志则 sign=空格，否则置 0
49  c=(type & ZEROPAD) ? '0':' ';
50  if(type & SIGN && num<0){
51  sign='-';
52  num-=num;
53  }
54  else
55  sign=(type & PLUS) ? '+':((type & SPACE) ? ' ':0);
56 // 若带符号，则宽度值减 1.若类型指出是特殊转换，则对于十六进制宽度再减少 2 位(用于 0x)，对于
57 // 八进制宽度减 1 (用于八进制转换结果)
58  if(sign) size--;
59  if(type & SPECIAL)
60  if(base==16) size-=2;
61  else if(base==8) size--;
62 // 如果数值 num 为 0，则临时字符串='0'；否则根据给定的基数将数值 num 转换成字符形式
63  i;
64  if(num==0)
65  tmp[i++]='0';
66  else while(num!=0)
67  tmp[i++]=digits[do_div(num,base)];
68 // 若数值字符个数大于精读值，则精度值扩展为数字个数值。宽度值减去用于存放数值字符的个数。
69  if(i>precision) precision=i;
70  size-=precision;
71 // 从这里真正开始形成所需要的转换结果，并暂时放在字符串 str 中。若类型中没有填零(ZEROPAD)
72 // 和左调整标志，则在 str 中首先填放剩余宽度值指出的空格数。若需带符号位，则存入符号。
73  if(!(type & (ZEROPAD+LEFT)))
74  while(size-->0)
75  *str++=' ';
76  if(sign)
77  *str++=sign;
78 // 若类型指出是特殊转换，则对于八进制转换结果头一位放置一个 '0';而对于十六进制则存放 '0x'.
79  if(type & SPECIAL)
80  if(base==8)
81  *str++='0';
82  else if(base==16){
83  *str++='0';
84  *str++=digits[33]; // 'X' 或 'x'
85  }
86 // 若类型中没有左调整(左靠齐)标志，则在剩余宽度中存放 c 字符('0' 或空格)，见 49 行。
87  if(!(type & LEFT))
88  while(size-->0)
89  *str++=c;
90 // 此时 i 存有数值 num 的数字个数。若数字个数小于精读值，则 str 中放入(精度值 - i)个 '0'.
91  while(i<precision--)
92  *str++='0';
93  while(i-->0) // 将转换好的数字字符填入 str 中。共 i 个。
94  *str++=tmp[i];
95 // 若宽度值仍大于零，则表示类型标志中有左靠齐标志标志。则在剩余宽度中放入空格。
96  while(size-->0)
97  *str++=' ';
98  return str; // 返回转换好的字符串。
99 }

107  char * s;
108  int * ip;
109 
110  int flags; // flags to number()
111  // number() 函数使用的标志
112  int field_width; // width of output field 输出字符宽度
113  int precision; // min. # of digits for integers;max number of chars for from
114  // fields. min.整数数字个数；max. 字符串中字符个数
115  int qualifier; // 'h','l',or 'L' for integer fields 'h','l',或 'L' 用于整数字段
116 // 首先将字符指针指向 buf，然后扫描格式字符串，对各个格式转换只是进行相应的处理
117  for(str=buf;*fmt;++fmt){
118 // 格式转换指示字符串均以 '%' 开始，这里从 fmt 格式字符串中扫描 '%'，寻找格式转换字符串的开始。
119 // 不是格式指示的一般字符均被依次存入 str
120  if(*fmt!='%'){
121  *str++=*fmt;
122  continue;
123  }
124 // 下面取得格式指示字符串中的标志域，并将标志常量放入 flags 变量中
125  // prcess flags.
126  flags=0;
127  repeat:
128  ++fmt; // this also skips first '%'
129  switch(*fmt){
130  case '-': flags|=LEFT; goto repeat; // 左靠齐调整
131  case '+': flags|=PLUS; goto repeat; // 放加号
132  case ' ': flags|=SPACE; goto repeat; // 放空格
133  case '#': flags|=SPECIAL; goto repeat; // 是特殊转换
134  case '0': flags|=ZEROPAD; goto repeat; //要填零(即 ‘0’)
135  }
136 // 去参数字段宽度阈值放入 field_width 变量中。若宽度域中是数值则直接取其为宽度值。若宽度域是
137 // 字符 '*' ，表示下一参数指定宽度，调用 va_arg 取宽度值。若此时宽度值 < 0，则该负数表示其带有
138 // 标志域 '-' 标志(左靠齐)，因此需在标志变量中添入该标志，并将字段宽度值取为其绝对值。
139  // get field field_width
140  field_width=-1;
141  if(is_digit(*fmt))
142  field_width=skip_atoi(&fmt);
143  else if(*fmt=='*'){
144  // it's the next argument 这里应该插入 ++fmt;
145  field_width=va_arg(args,int);
146  if(field_width<0){
147  field_width=-field_width;
148  flags|=LEFT;
149  }
150  }
151 // 下面代码取格式转换串精度域，并放入 precision 变量中。精度域的开始标志是 '.'。其处理过程与上面
152 // 宽度域类似。若精度域中是数值则直接取其为精度值。若精度域中是字符 '*'，表示下一个参数指定精度。

154  // get the precision
155  precision=-1;
156  if(*fmt=='.'){
157  ++fmt;
158  if(is_digit(*fmt))
159  precision=skip_atoi(&fmt);
160  else if(*fmt=='*'){
161  // it's the next argument
162  precision=va_arg(args,int);
163  }
164  if(precision<0)
165  precision=0;
166  }

168  // get the conversion qualifier
169  qualifier=-1;
170  if(*fmt=='h'||*fmt=='l'||*fmt=='L'){
171  qualifier=*fmt;
172  ++fmt;
173  }
174 // 下面分析转换指示符。如果转换指示符是 'c'，则表示对应参数应是字符。此时如果标志域表明不是左
175 // 对齐，则该字段前面放入宽度域值 -1 个空格字符，然后再被放入参数字符。如果宽度域还大于 0，

187 // 拓展精度域=字符串长度。此时如果标志域表明不是左靠齐，则该字段前放入(宽度值-字符串长度)个空格
188 // 字符。然后再放入参数字符串。如果宽度域还大于 0，则表示为左靠齐，则在参数字符串后面添加(宽度值-

195  else if(len>precision)
